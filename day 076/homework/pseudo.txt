sumDigits(num){
    num = str(num) //ვაქცევ Num-ს სტრინგად რათგან length-მა იმუშაოს
    result = 0 //default პასუხი არის 0//
    for i in range(num.length){ //ციკლს ვქმნი რომელიც განმეორდება იმდეჯნერ თუ რამხელაც არის გადმოცემული რიცხვი//
        result = result + int(num[i]) //რათგან I თითო ციკლზე იმატებს 1-ს რეზულტს დავამატებ i ინდექსზე მდგომ რიცხვის ინტეჯერს. სიტყვაზე: თუ გადმოცემული გვაქვს 12, ჯერ რესულტს დავამატებ i-ზე მდგომ ციფრს, (0 იდნექსზე ანუ 1-ს) და შემდეგ რათგანაც i იმატებს თითო ციკლზე, დაემატება 1 ინდექსზე მდგომი რიცხვი რეზულტს.(ანუ 2)
    }
    return result //ვაბრუნებ რეზულტს
}
